Basic Cytoscape.js Graph

        <div id="cy"></div>
        <script>
        let cy = cytoscape({
            container: document.getElementById('cy'),
            elements: [
                { data: { id: 'A', label: 'Node A' } },
                { data: { id: 'B', label: 'Node B' } },
                { data: { id: 'C', label: 'Node C' } },
                { data: { id: 'D', label: 'Node D' } },
                { data: { id: 'E', label: 'Node E' } },
                { data: { id: 'AB', source: 'A', target: 'B' } },
                { data: { id: 'BC', source: 'B', target: 'C' } },
                { data: { id: 'CD', source: 'C', target: 'D' } },
                { data: { id: 'DE', source: 'D', target: 'E' } },
                { data: { id: 'EA', source: 'E', target: 'A' } }
            ],
            style: [
                { selector: 'node', style: { 'background-color': '#007bff', 'label': 'data(label)', 'text-valign': 'center', 'color': 'white' } },
                { selector: 'edge', style: { 'line-color': '#999', 'width': 2, 'curve-style': 'bezier' } },
                { selector: '.highlighted', style: { 'background-color': 'yellow' } }
            ],
            layout: { name: 'breadthfirst' }
        });
        
        cy.on('tap', 'node', function(evt) {
            let node = evt.target;
            cy.elements().removeClass('highlighted');
            node.addClass('highlighted');
        });
        </script>

This example initializes a simple graph with 5 nodes and 5 edges. Clicking on a node highlights it.
